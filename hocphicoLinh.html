async function copyTableAsImage() { try { const element = document.getElementById('captureArea'); const canvas = await html2canvas(element, { scale: window.devicePixelRatio || 2, useCORS: true, backgroundColor: null }); canvas.toBlob(async function (blob) { await navigator.clipboard.write([ new ClipboardItem({ [blob.type]: blob }) ]); alert('✅ Đã sao chép ảnh chất lượng cao vào clipboard!'); }, 'image/png'); } catch (error) { console.error('❌ Lỗi sao chép:', error); alert('Không thể sao chép ảnh. Vui lòng thử lại hoặc chụp màn hình.'); } }
